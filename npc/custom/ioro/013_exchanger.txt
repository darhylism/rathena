poring_w02,82,101,4	script	Coupon Exchanger	4_F_PINKWOMAN,{

    set @name$,"[^FF0000Coupon Exchanger^000000]";
    mes @name$;
    mes "What do you want to do?";
    next;
    switch(select("Information","Convert berry","Nothing")) {

        case 1:
            mes @name$;
            mes "I can convert your Coupon/s into usable items and vice versa.";
            mes "1 Coupon = 400 Yggdrasil Berry";
            close;

        case 2:
            switch(select("Berry to Coupon","Coupon to Berry")) {

                case 1:
                    if(countitem(607) < 400) goto noBerry;
                    if (checkweight(35043,1) == 0 ) goto overWeight;
                    delitem 607,400;
                    mes @name$;
                    mes "There you go. Hope to see you again!";
                    getitem 35043,1;
                    close;

                case 2:
                    if(countitem(35043) < 1) goto noCoupon;
                    if (checkweight(607,400) == 0 ) goto overWeight;
                    delitem 35043,1;
                    mes @name$;
                    mes "There you go. Hope to see you again!";
                    getitem 607,400;
                    close;
			}
		default:
			close;
            }
            end;
noBerry:
    mes @name$;
    mes "I'm Sorry but you don't have enough items.";
    close;

overWeight:
    mes @name$;
    mes "Sorry you're overweight.";
    close;

noCoupon:
    mes @name$;
    mes "I'm Sorry but you don't have Coupons.";
    close;
	
// OnInit:
        // waitingroom "Coupon Trader",0;  // WITH WAITING ROOM
end;
}
